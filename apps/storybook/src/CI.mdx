export function getText(index, color) {
    let res = ""

    if (index >= 3) {
      res += "On "
    }

    res += color;

    if (index % 3 === 0) {
      // pass
    } else if (index % 3 === 1) {
      res += " Hover"
    } else {
      res += " Container"
    }

    return res;

}

            {getText(index, colorName)}

}
