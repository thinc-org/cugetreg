export const ColorGrid = ({colorName, classNames}) => {
  function getText(index, color) {
    let res = ""

    if (index >= 3) {
      res += "On "
    }

    res += color;

    if (index % 3 === 0) {
      // pass
    } else if (index % 3 === 1) {
      res += " Hover"
    } else {
      res += " Container"
    }

    return res;
  }

  return (
    <div className="grid grid-cols-3">
      {classNames.map((className, index) => (
        <div className={`h-16 ${className[0]}`} title={className[0]}>
          <p className={`text-center !font-sans ${className[1]}`}>
            {getText(index, colorName)}
          </p>
        </div>
      ))}
    </div>
  )
}
